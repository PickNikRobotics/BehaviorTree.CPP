90/**
91 * @brief Machine-readable cause for why an input port could not be read.
92 *
93 * The plain string error returned by TreeNode::getInputStamped() is fine for
94 * console messages, but callers that want to react differently to different
95 * failure modes (for example, "the user wired a blackboard key but forgot to
96 * populate it" versus "the port is simply unwired and the behavior should
97 * fall back to its default") cannot tell those cases apart from a string.
98 *
99 * TreeNode::getInputStampedWithDiagnostic() returns this enum alongside the
100 * original human-readable message so upstream code can make that distinction.
101 */
102enum class PortError : uint8_t
103{
104 ManifestMissing, ///< config().manifest is null and the port has no XML entry.
105 ManifestKeyMissing, ///< The key is not in the manifest and not in the XML.
106 NoDefaultNoWiring, ///< The manifest has the key but no default value and the XML did not wire it.
107 BlackboardKeyNotFound, ///< The port is wired to {foo} but the blackboard has no entry called foo.
108 BlackboardEntryEmpty, ///< The blackboard entry exists but its value is empty.
109 InvalidBlackboard, ///< config().blackboard is null.
110 ConversionFailed, ///< parseString<T>() threw while converting a string to T.
111 CastFailed, ///< Any::cast<T>() or a vector element cast threw.
112};

149/**
150 * convertFromString is used to convert a string into a custom type.
151 *
152 * This function is invoked under the hood by TreeNode::getInput(), but only when the
153 * input port contains a string.
154 *
155 * If you have a custom type, you need to implement the corresponding
156 * template specialization.
157 *
158 * If the string starts with the prefix "json:", it will
159 * fall back to convertFromJSON()
160 */
161template <typename T>
162[[nodiscard]] inline T convertFromString(StringView str)
163{
164 // if string starts with "json:{", try to parse it as json
165 if(StartWith(str, "json:"))
166 {
167 str.remove_prefix(5);
168 return convertFromJSON<T>(str);
169 }
170
171 auto type_name = BT::demangle(typeid(T));
172
173 std::cerr << "You (maybe indirectly) called BT::convertFromString() for type ["
174 << type_name << "], but I can't find the template specialization.\n"
175 << std::endl;
176
177 throw LogicError(std::string("You didn't implement the template specialization of "
178 "convertFromString for this type: ") +
179 type_name);
180}

301/**
302 * @brief toStr is the reverse operation of convertFromString.
303 *
304 * If T is a custom type and there is no template specialization,
305 * it will try to fall back to toJsonString()
306 */
307template <typename T>
308[[nodiscard]] std::string toStr(const T& value)
309{
310 auto throw_unspecialized_error = []() {
311 throw LogicError(StrCat("Function BT::toStr<T>() not specialized for type [",
312 BT::demangle(typeid(T)), "]"));
313 };
314
315 if constexpr(IsConvertibleToString<T>())
316 {
317 return static_cast<std::string>(value);
318 }
319 else if constexpr(is_vector<T>::value)
320 {
321 try
322 {
323 if(value.empty())
324 {
325 return "";
326 }
327 using InnerType = typename is_vector<T>::ValueType;
328 std::stringstream ss;
329 for(auto it = value.begin(); it != --value.end(); ++it)
330 {
331 ss << toStr<InnerType>(*it) << ';';
332 }
333 ss << toStr<InnerType>(value.back());
334 return ss.str();
335 }
336 catch(LogicError&)
337 {
338 throw_unspecialized_error();
339 }
340 }
341 else if constexpr(!std::is_arithmetic_v<T>)
342 {
343 if(auto str = toJsonString(Any(value)))
344 {
345 return *str;
346 }
347
348 throw_unspecialized_error();
349 }
350 else
351 {
352 return std::to_string(value);
353 }
354}

522template <typename T = AnyTypeAllowed>
523[[nodiscard]] std::pair<std::string, PortInfo> CreatePort(PortDirection direction,
524 StringView name,
525 StringView description = {})
526{
527 auto sname = static_cast<std::string>(name);
528 if(!IsAllowedPortName(sname))
529 {
530 throw RuntimeError("The name of a port must not be `name` or `ID` "
531 "and must start with an alphabetic character. "
532 "Underscore is reserved.");
533 }
534 ThrowIfPortNameContainsWhitespace(sname);
535
536 std::pair<std::string, PortInfo> out;
537
538 if(std::is_same<T, void>::value)
539 {
540 out = { sname, PortInfo(direction) };
541 }
542 else
543 {
544 out = { sname, PortInfo(direction, typeid(T), GetAnyFromStringFunctor<T>()) };
545 }
546 if(!description.empty())
547 {
548 out.second.setDescription(description);
549 }
550 return out;
551}
